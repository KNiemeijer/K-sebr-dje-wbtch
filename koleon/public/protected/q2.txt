/*QUERY OM ALLE PRODUCTEN TE LADEN*/

SELECT P.ProductName, P.Description, P.Price, P.Stock, U.Username, C.CategoryName, 
SCG.SubCategoryName, SCG2.SubCategoryName, M.ManufacturerName 
FROM 
PRODUCTS AS P INNER JOIN CATEGORIES AS C ON P.CategoryID=C.CategoryID
INNER JOIN SUBCATEGORIES AS SCG ON P.SubCategoryID=SCG.SubCategoryID
INNER JOIN SUBCATEGORIES AS SCG2 ON P.SubCategory2ID=SCG2.SubCategoryID
INNER JOIN MANUFACTURERS AS M ON P.ManufacturerID=M.ManufacturerID
INNER JOIN USERS AS U ON P.SellerID=U.UserID
;
